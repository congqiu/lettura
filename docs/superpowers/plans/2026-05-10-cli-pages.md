> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 lettura-cli 添加 `pages` 嵌套子命令，支持 publish/list/update/delete/restore/share 六个操作。

**Architecture:** 在 CLI crate 中新增 `pages` 命令模块，复用现有 ApiClient 模式并扩展 multipart 上传能力。`pages publish` 合并 upload+create 为单步操作，支持文件/目录/URL 三种输入。`pages update --files` 自动完成上传+替换。

**Tech Stack:** Rust, clap 4 (derive), reqwest (multipart), zip crate (目录打包), serde_json

```rust
pub async fn upload_files(&self, file_path: &Path) -> Result<UploadResponse, CliError> {
    let file_name = file_path
        .file_name()
        .unwrap_or_default()
        .to_string_lossy()
        .to_string();
    let file_bytes = tokio::fs::read(file_path).await.map_err(|e| {
        CliError::UploadFailed(format!("Failed to read file: {e}"))
    })?;

    let part = reqwest::multipart::Part::bytes(file_bytes)
        .file_name(file_name)
        .mime_str("application/octet-stream")
        .map_err(|e| CliError::UploadFailed(format!("MIME error: {e}")))?;

    let form = reqwest::multipart::Form::new().part("files", part);

    let url = format!("{}/api/v1/pages/upload", self.base_url);
    let resp = self
        .http
        .post(&url)
        .bearer_auth(&self.token)
        .multipart(form)
        .send()
        .await
        .map_err(|e| CliError::RequestFailed(e.to_string()))?;

    let status = resp.status();
    if !status.is_success() {
        let body = resp.text().await.unwrap_or_default();
        return Err(CliError::UploadFailed(format!(
            "Server returned {status}: {body}"
        )));
    }

    resp.json::<UploadResponse>()
        .await
        .map_err(|e| CliError::ParseFailed(e.to_string()))
}
```

```rust
use std::io::Write;
use std::path::Path;

use crate::api_types::{PageListResponse, PageResponse, PageShareResponse, UploadResponse};
use crate::cli::{
    PagesCmd, PagesDeleteArgs, PagesListArgs, PagesPublishArgs, PagesRestoreArgs,
    PagesShareArgs, PagesUpdateArgs,
};
use crate::client::ApiClient;
use crate::error::CliError;
use crate::output::OutputFormat;

pub async fn run(cmd: &PagesCmd, client: &ApiClient, output: OutputFormat) -> Result<(), CliError> {
    match cmd {
        PagesCmd::Publish(args) => publish(args, client, output).await,
        PagesCmd::List(args) => list(args, client, output).await,
        PagesCmd::Update(args) => update(args, client, output).await,
        PagesCmd::Delete(args) => delete(args, client, output).await,
        PagesCmd::Restore(args) => restore(args, client, output).await,
        PagesCmd::Share(args) => share(args, client, output).await,
    }
}

async fn publish(
    args: &PagesPublishArgs,
    client: &ApiClient,
    output: OutputFormat,
) -> Result<(), CliError> {
    let file_path = prepare_upload_file(&args.source)?;
    let upload: UploadResponse = client.upload_files(&file_path).await?;

    let title = args
        .title
        .as_deref()
        .or(upload.suggested_title.as_deref())
        .unwrap_or("Untitled");

    let mut body = serde_json::json!({
        "upload_id": upload.upload_id,
        "entry_file": args.entry_file.as_deref().unwrap_or(&upload.default_entry),
        "title": title,
    });

    if let Some(desc) = &args.description {
        body["description"] = serde_json::Value::String(desc.clone());
    }
    if let Some(pw) = &args.password {
        body["password"] = serde_json::Value::String(pw.clone());
    }
    if let Some(exp) = &args.expires_at {
        body["expires_at"] = serde_json::Value::String(exp.clone());
    }

    let page: PageResponse = client.post("/api/v1/pages", &body).await?;

    match output {
        OutputFormat::Json => println!("{}", serde_json::to_string_pretty(&page)?),
        _ => println!("Published: {} → {}", page.title, page.url),
    }

    Ok(())
}

async fn list(
    args: &PagesListArgs,
    client: &ApiClient,
    output: OutputFormat,
) -> Result<(), CliError> {
    let path = format!(
        "/api/v1/pages?status={}&page={}&limit={}",
        args.status, args.page, args.limit
    );
    let result: PageListResponse = client.get(&path).await?;

    match output {
        OutputFormat::Json => println!("{}", serde_json::to_string_pretty(&result)?),
        OutputFormat::Ids => {
            for item in &result.items {
                println!("{}", item.id);
            }
        }
        OutputFormat::Human => {
            for item in &result.items {
                println!(
                    "{}\t{}\t{}\t{}\t{}",
                    item.slug, item.title, item.status, item.url, item.created_at
                );
            }
            eprintln!(
                "Total: {} (page {}/{})",
                result.total,
                result.page,
                (result.total + result.limit - 1) / result.limit
            );
        }
    }

    Ok(())
}

async fn update(
    args: &PagesUpdateArgs,
    client: &ApiClient,
    output: OutputFormat,
) -> Result<(), CliError> {
    if args.password.is_some() && args.clear_password {
        return Err(CliError::InvalidArgs(
            "--password and --clear-password are mutually exclusive".into(),
        ));
    }

    let mut body = serde_json::json!({});

    if let Some(title) = &args.title {
        body["title"] = serde_json::Value::String(title.clone());
    }
    if let Some(desc) = &args.description {
        body["description"] = serde_json::Value::String(desc.clone());
    }
    if let Some(pw) = &args.password {
        body["password"] = serde_json::Value::String(pw.clone());
    }
    if args.clear_password {
        body["password"] = serde_json::Value::Null;
    }
    if let Some(status) = &args.status {
        body["status"] = serde_json::Value::String(status.clone());
    }
    if let Some(exp) = &args.expires_at {
        if exp == "none" {
            body["expires_at"] = serde_json::Value::Null;
        } else {
            body["expires_at"] = serde_json::Value::String(exp.clone());
        }
    }

    if let Some(files_source) = &args.files {
        let file_path = prepare_upload_file(files_source)?;
        let upload: UploadResponse = client.upload_files(&file_path).await?;
        body["upload_id"] = serde_json::Value::String(upload.upload_id);
        body["entry_file"] = serde_json::Value::String(
            args.entry_file
                .as_deref()
                .unwrap_or(&upload.default_entry)
                .to_string(),
        );
    } else if let Some(entry) = &args.entry_file {
        body["entry_file"] = serde_json::Value::String(entry.clone());
    }

    let path = format!("/api/v1/pages/{}", args.id);
    let page: PageResponse = client.http_patch(&path, &body).await?;

    match output {
        OutputFormat::Json => println!("{}", serde_json::to_string_pretty(&page)?),
        _ => println!("Updated: {}", page.title),
    }

    Ok(())
}

/// Prepare a file for upload from a source path or URL.
fn prepare_upload_file(source: &str) -> Result<std::path::PathBuf, CliError> {
    if source.starts_with("http://") || source.starts_with("https://") {
        return fetch_url_to_temp_file(source);
    }

    let path = Path::new(source);

    if path.is_dir() {
        return zip_directory(path);
    }

    if path.is_file() {
        return Ok(path.to_path_buf());
    }

    Err(CliError::InvalidArgs(format!(
        "Path does not exist: {source}"
    )))
}

fn fetch_url_to_temp_file(url: &str) -> Result<std::path::PathBuf, CliError> {
    let rt = tokio::runtime::Handle::current();
    let response = rt.block_on(async {
        reqwest::Client::new()
            .get(url)
            .timeout(std::time::Duration::from_secs(30))
            .send()
            .await
            .map_err(|e| CliError::UploadFailed(format!("Failed to fetch URL: {e}")))
    })?;

    let status = response.status();
    if !status.is_success() {
        return Err(CliError::UploadFailed(format!(
            "URL returned {status}"
        )));
    }

    let content_type = response
        .headers()
        .get("content-type")
        .and_then(|v| v.to_str().ok())
        .unwrap_or("");
    if !content_type.contains("text/html") {
        return Err(CliError::UploadFailed(format!(
            "URL content is not HTML (content-type: {content_type})"
        )));
    }

    let bytes = rt.block_on(async {
        response
            .bytes()
            .await
            .map_err(|e| CliError::UploadFailed(format!("Failed to read response: {e}")))
    })?;

    if bytes.len() > 10 * 1024 * 1024 {
        return Err(CliError::UploadFailed(
            "Fetched content exceeds 10MB limit".into(),
        ));
    }

    let temp_dir = std::env::temp_dir().join("lettura-cli-pages");
    std::fs::create_dir_all(&temp_dir).map_err(|e| {
        CliError::UploadFailed(format!("Failed to create temp dir: {e}"))
    })?;

    let temp_file = temp_dir.join(format!("url-{}.html", uuid::Uuid::new_v4()));
    let mut f = std::fs::File::create(&temp_file).map_err(|e| {
        CliError::UploadFailed(format!("Failed to create temp file: {e}"))
    })?;
    f.write_all(&bytes).map_err(|e| {
        CliError::UploadFailed(format!("Failed to write temp file: {e}"))
    })?;

    Ok(temp_file)
}

fn zip_directory(dir: &Path) -> Result<std::path::PathBuf, CliError> {
    let temp_dir = std::env::temp_dir().join("lettura-cli-pages");
    std::fs::create_dir_all(&temp_dir).map_err(|e| {
        CliError::UploadFailed(format!("Failed to create temp dir: {e}"))
    })?;

    let zip_path = temp_dir.join(format!("dir-{}.zip", uuid::Uuid::new_v4()));
    let zip_file = std::fs::File::create(&zip_path).map_err(|e| {
        CliError::UploadFailed(format!("Failed to create zip file: {e}"))
    })?;
    let mut zip_writer = zip::ZipWriter::new(zip_file);
    let options = zip::write::SimpleFileOptions::default()
        .compression_method(zip::CompressionMethod::Deflated);

    let mut file_count = 0u32;
    collect_and_zip(dir, dir, &mut zip_writer, options, &mut file_count)?;

    if file_count == 0 {
        std::fs::remove_file(&zip_path).ok();
        return Err(CliError::UploadFailed(
            "Directory is empty (after excluding hidden files)".into(),
        ));
    }

    zip_writer.finish().map_err(|e| {
        CliError::UploadFailed(format!("Failed to finalize zip: {e}"))
    })?;

    let zip_size = std::fs::metadata(&zip_path)
        .map(|m| m.len())
        .unwrap_or(0);
    if zip_size > 50 * 1024 * 1024 {
        std::fs::remove_file(&zip_path).ok();
        return Err(CliError::UploadFailed(
            "Zipped directory exceeds 50MB limit".into(),
        ));
    }

    Ok(zip_path)
}

fn collect_and_zip(
    base: &Path,
    current: &Path,
    zip_writer: &mut zip::ZipWriter<std::fs::File>,
    options: zip::write::SimpleFileOptions,
    file_count: &mut u32,
) -> Result<(), CliError> {
    let entries = std::fs::read_dir(current).map_err(|e| {
        CliError::UploadFailed(format!("Failed to read directory: {e}"))
    })?;

    for entry in entries {
        let entry = entry.map_err(|e| {
            CliError::UploadFailed(format!("Failed to read dir entry: {e}"))
        })?;
        let name = entry.file_name();
        let name_str = name.to_string_lossy();

        if name_str.starts_with('.') || name_str == "__MACOSX" {
            continue;
        }

        let path = entry.path();
        let relative = path
            .strip_prefix(base)
            .unwrap_or(&path)
            .to_string_lossy()
            .to_string();

        if path.is_dir() {
            zip_writer
                .add_directory(&relative, options)
                .map_err(|e| {
                    CliError::UploadFailed(format!("Failed to add dir to zip: {e}"))
                })?;
            collect_and_zip(base, &path, zip_writer, options, file_count)?;
        } else {
            let mut f = std::fs::File::open(&path).map_err(|e| {
                CliError::UploadFailed(format!("Failed to open file: {e}"))
            })?;
            zip_writer
                .start_file(&relative, options)
                .map_err(|e| {
                    CliError::UploadFailed(format!("Failed to add file to zip: {e}"))
                })?;
            std::io::Copy::copy(&mut f, zip_writer).map_err(|e| {
                CliError::UploadFailed(format!("Failed to write file to zip: {e}"))
            })?;
            *file_count += 1;
        }
    }

    Ok(())
}
```
